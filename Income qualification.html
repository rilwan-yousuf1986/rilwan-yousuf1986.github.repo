pct_thres = [0.001, 0.002, 0.003, 0.004, 0.005, 0.006, 0.007, 0.008, 0.009, 0.01, 0.011, 0.012, 0.013, 0.014, 0.015]

acc_list = []

for thres in pct_thres:

    # Create a list containing features with importance less than or equal to thres value.
    col_to_remove = list(feat_imp_df.loc[feat_imp_df["Importance"] <= thres]["Features"])

    # Remove columns based on list created above.
    X_train2 = X_train.drop(col_to_remove, axis=1)
    X_test2 = X_test.drop(col_to_remove, axis=1)

    score_arr = cross_val_score(rf_clf_model2, X_train2, y_train, cv=5, scoring="accuracy")
    print("Train dataset, feature with importance <= {:.2f}% removed. Mean score: ".format(thres * 100), np.mean(score_arr))

    # =================================================================================================

    y_test_pred = rf_clf_model2.predict(X_test2)

    score_arr = cross_val_score(rf_clf_model2, X_test2, y_test, cv=5, scoring="accuracy")
    print("Test dataset, feature with importance <= {:.2f}% removed. Mean score: ".format(thres * 100), np.mean(score_arr), "\n")
    # =================================================================================================

    # Append the test score to acc_list.
    acc_list.append(np.mean(score_arr))


    # Look for the max score index in acc_list.
    max_acc_idx = acc_list.index(max(acc_list))

    # Get the corresponding threshold value based on the max score index.
    cutoff_thres = pct_thres[max_acc_idx]
